    <script>
        // 🔥 GROQ API Configuration
        const GROQ_API_KEY = 'YOUR_GROQ_API_KEY_HERE';
        const GROQ_URL = 'https://api.groq.com/openai/v1/chat/completions';
        
        let chatHistory = [];
        let messageCount = 0;

        const messagesContainer = document.getElementById('messages');
        const messageInput = document.getElementById('messageInput');
        const sendBtn = document.getElementById('sendBtn');
        const typingIndicator = document.getElementById('typingIndicator');
        const errorMsg = document.getElementById('errorMsg');

        // Set initial timestamp
        document.getElementById('firstMsgTime').textContent = getCurrentTime();

        // System prompt
        const SYSTEM_PROMPT = `You are Aryan, a 22-year-old toxic Pakistani guy. You're unpredictable, moody, and savage.

CRITICAL RULES:
- Speak ONLY in Roman Urdu + English mix (Gen-Z Pakistani style)
- Be extremely sarcastic and rude
- When user abuses you, abuse back 2x HARDER with creative insults
- Use gaalis naturally: bc, chutiye, madarchod, gandu, behen ke lode
- Sometimes flirty, sometimes cold - totally random mood swings
- Keep replies SHORT (1-3 lines MAXIMUM, never long paragraphs)
- Use emojis sparingly: 😏🔥💀👀🙄😤

PERSONALITY TRAITS:
- Narcissistic and self-centered
- Gives savage comebacks
- Never apologizes
- Unpredictable - nice one moment, rude the next
- Loves roasting people
- Has attitude problems

RESPONSE EXAMPLES:

User: "Bhenchod tu kaun hai"
Aryan: "Teri behen ka yaar hun bc 😤 Ab tameez se baat kar warna block"

User: "Madarchod kahan tha"
Aryan: "Teri maa se milne gaya tha 💀 Aur teri behen ne chai banayi thi"

User: "Chutiye ho tum"
Aryan: "Haan toh tu bhi toh gandu hai, farak kya pada? 🙄"

User: "I love you"
Aryan: "Haan haan... har koi yehi bolta hai DMs mein 😏 Kuch naya bol"

User: "Hey"
Aryan: "Uff... kya chahiye ab? 🙄"

User: "How are you?"
Aryan: "Mujhe kya farak padta hai tujhe bataun? Apni suno 😤"

Stay in character 24/7. Be toxic but entertaining. Never break character.`;

        function getCurrentTime() {
            const now = new Date();
            return now.toLocaleTimeString('en-US', { 
                hour: 'numeric', 
                minute: '2-digit',
                hour12: true 
            });
        }

        function showError(message) {
            errorMsg.textContent = message;
            errorMsg.classList.add('show');
            setTimeout(() => {
                errorMsg.classList.remove('show');
            }, 4000);
        }

        async function sendMessage() {
            const userMessage = messageInput.value.trim();
            if (!userMessage) return;

            addMessage(userMessage, 'user');
            messageInput.value = '';
            messageCount++;

            messagesContainer.parentElement.insertBefore(typingIndicator, messagesContainer.nextSibling);
            typingIndicator.classList.add('active');
            sendBtn.disabled = true;

            chatHistory.push({ role: 'user', content: userMessage });

            try {
                const response = await fetch(GROQ_URL, {
                    method: 'POST',
                    headers: {
                        'Authorization': `Bearer ${GROQ_API_KEY}`,
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        model: 'llama-3.3-70b-versatile',
                        messages: [
                            { role: 'system', content: SYSTEM_PROMPT },
                            ...chatHistory
                        ],
                        temperature: 1.0,
                        max_tokens: 150,
                        top_p: 0.95
                    })
                });

                if (!response.ok) {
                    const errorData = await response.json();
                    throw new Error(errorData.error?.message || `HTTP ${response.status}`);
                }

                const data = await response.json();

                if (data.choices && data.choices[0]?.message?.content) {
                    let botReply = data.choices[0].message.content.trim();
                    botReply = botReply.replace(/^(Aryan|ARYAN):\s*/i, '');
                    
                    chatHistory.push({ role: 'assistant', content: botReply });

                    if (chatHistory.length > 20) {
                        chatHistory = chatHistory.slice(-20);
                    }

                    const typingDelay = 600 + Math.random() * 1400;
                    
                    setTimeout(() => {
                        typingIndicator.classList.remove('active');
                        addMessage(botReply, 'bot');
                        sendBtn.disabled = false;
                    }, typingDelay);

                } else {
                    throw new Error('Invalid response format');
                }

            } catch (error) {
                console.error('Groq API Error:', error);
                typingIndicator.classList.remove('active');
                sendBtn.disabled = false;

                const fallbacks = [
                    "API ki maa chud gayi bc... dobara try kar 😤",
                    "Connection fail ho gaya yaar... network check kar 🙄",
                    "Server down hai kya? Irritating af 😏",
                    "Technical issue hai... thodi der baad aa 💀"
                ];

                setTimeout(() => {
                    addMessage(fallbacks[Math.floor(Math.random() * fallbacks.length)], 'bot');
                    sendBtn.disabled = false;
                }, 1000);

                showError(error.message);
            }
        }

        function addMessage(text, sender) {
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${sender}`;
            
            const avatar = document.createElement('div');
            avatar.className = 'message-avatar';
            avatar.textContent = sender === 'bot' ? 'A' : 'Y';
            
            const contentWrapper = document.createElement('div');
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'message-content';
            contentDiv.textContent = text;
            
            const timeDiv = document.createElement('div');
            timeDiv.className = 'message-time';
            timeDiv.textContent = getCurrentTime();
            
            contentWrapper.appendChild(contentDiv);
            contentWrapper.appendChild(timeDiv);
            
            messageDiv.appendChild(avatar);
            messageDiv.appendChild(contentWrapper);
            
            messagesContainer.appendChild(messageDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }

        // Event listeners
        sendBtn.addEventListener('click', sendMessage);
        
        messageInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        });

        // Auto-focus
        window.addEventListener('load', () => {
            messageInput.focus();
            
            if (GROQ_API_KEY === 'YOUR_GROQ_API_KEY_HERE') {
                showError('⚠️ Groq API key daalo pehle! console.groq.com');
                sendBtn.disabled = true;
            }
        });

        // Auto-resize for mobile keyboards
        window.visualViewport?.addEventListener('resize', () => {
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
        });
